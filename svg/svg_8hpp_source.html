51  inline std::vector<Point> convex_hull(std::vector<Point>& points) {
58  if (points.size() < 3) return {}; // Need at least three points
59  std::vector<Point> hull;
60 
61  // Find leftmost point (ties don't matter)
62  int left = 0;
63  for (size_t i = 0; i < points.size(); i++)
64  if (points[i].first < points[left].first) left = (int)i;
65 
66  // While we don't reach leftmost point
67  int current = left, next;
68  do {
69  // Add to convex hull
70  hull.push_back(points[current]);
71 
72  // Keep moving counterclockwise
73  next = (current + 1) % points.size();
74  for (size_t i = 0; i < points.size(); i++) {
75  // We've found a more counterclockwise point --> update next
76  if (orientation(points[current], points[next], points[i]) == COUNTERCLOCKWISE)
77  next = (int)i;
78  }
79 
80  current = next;
81  } while (current != left);
82 
83  return hull;
84  }

461  inline std::pair<double, double> Line::along(double percent) {
466  double x_pos, y_pos;
467 
468  if (x1() != x2()) {
469  double length = percent * this->get_length();
470  double discrim = std::sqrt(4 * pow(length, 2) * (1 / (1 + pow(get_slope(), 2))));
471 
472  double x_a = (2 * x1() + discrim) / 2;
473  double x_b = (2 * x1() - discrim) / 2;
474  x_pos = x_a;
475 
476  if ((x_a > x1() && x_a > x2()) || (x_a < x1() && x_a < x2()))
477  x_pos = x_b;
478 
479  y_pos = get_slope() * (x_pos - x1()) + y1();
480  }
481  else { // Edge case:: Completely vertical lines
482  x_pos = x1();
483 
484  if (y1() > y2()) // Downward pointing
485  y_pos = y1() - percent * this->get_length();
486  else
487  y_pos = y1() + percent * this->get_length();
488  }
489 
490  return std::make_pair(x_pos, y_pos);
491  }
